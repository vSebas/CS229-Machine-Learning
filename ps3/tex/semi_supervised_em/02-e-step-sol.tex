\begin{answer}

Calculate soft guesses for latent variables \(z^{(i)}\) just for the unlabeled data points
$$
\begin{align}
w^{(i)}_j &= p(z^{(i)}=j|x^{(i)};\phi_j, \mu_j, \Sigma_j) \\
    &= \frac{p(x^{(i)}|z^{(i)}=j;\phi_j, \mu_j, \Sigma_j)\,p(z^{(i)}=j;\phi_j, \mu_j, \Sigma_j)}{\sum_{l=1}^k p(x^{(i)}|z^{(i)}=l;\phi_j, \mu_j, \Sigma_j)\,p(z^{(i)}=l;\phi_j, \mu_j, \Sigma_j)} \\
    &= \frac{\phi_j\,\mathcal{N}(x^{(i)}\!\mid\mu_j,\Sigma_j)}{\sum_{\ell=1}^k \phi_\ell \,\mathcal{N}(x^{(i)}\!\mid\mu_\ell,\Sigma_\ell)}
\end{align}
$$
where \(\mathcal{N}(x^{(i)}\!\mid\mu_j,\Sigma_j)\) is the Gaussian density function for component \(j\) evaluated at \(x^{(i)}\)
$$
\mathcal{N}(x^{(i)}\!\mid\mu_j,\Sigma_j) = \frac{1}{(2\pi)^{d/2}|\Sigma_j|^{1/2}} \exp\left(-\frac{1}{2}(x^{(i)}-\mu_j)^T \Sigma_j^{-1} (x^{(i)}-\mu_j)\right)
$$
\end{answer}
